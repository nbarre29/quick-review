@echo off
REM Set Java 21 Environment Variables and Run JSON Parser
REM This sets JAVA_HOME and updates PATH for the current command prompt session

echo Setting Java 21 environment...

REM Set JAVA_HOME to Java 21 installation directory
set JAVA_HOME=C:\jdk21

REM Add Java 21 bin directory to the beginning of PATH
set PATH=%JAVA_HOME%\bin;%PATH%

REM Display the Java version to confirm
echo.
echo Java environment set to:
java -version

echo.
echo JAVA_HOME is set to: %JAVA_HOME%
echo.

REM Run the JSON parser with findings.json
echo Running JSON parser on findings.json...
echo.
java -jar json-parser.jar findings.json

echo.
echo Execution complete.
echo.

REM Keep the window open
pause
